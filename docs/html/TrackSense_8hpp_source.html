119 static double
120 converter_lat_long(
121 const std::string& valor,
122 const std::string& hemisf
123 ){
124
125 if( valor.empty() ){ return 0.0; }

127 double valor_cru = std::stod(valor);
128 double graus = floor(valor_cru / 100);
129 double minutos = valor_cru - graus * 100;
130 double dec = (graus + minutos / 60.0) * ( (hemisf == "S" || hemisf == "W" ) ? -1 : 1 );
