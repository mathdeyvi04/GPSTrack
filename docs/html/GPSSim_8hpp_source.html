86 degrees_to_NMEA(
87 double graus_decimais,
88 bool is_lat,
89 std::string& ddmm,
90 char& hemisf
91 ){
92
93 hemisf = (is_lat) ? (
94 ( graus_decimais >= 0 ) ? 'N' : 'S'
95 ) :
96 (
97 ( graus_decimais >= 0 ) ? 'E' : 'W'
98 );
99
100 double valor_abs = std::fabs(graus_decimais);
101 int graus = static_cast<int>(std::floor(valor_abs));
102 double min = (valor_abs - graus) * 60.0;
103
104 // Não utilizamos apenas a função de formatar_inteiro, pois
105 // utilizaremos o oss em seguida.
106 std::ostringstream oss;
107 if(
108 is_lat
109 ){
110
111 oss << std::setw(2)
112 << std::setfill('0')
113 << graus
114 << std::fixed
115 << std::setprecision(4)
116 << std::setw(7)
117 << std::setfill('0')
118 << min;
119 }
120 else{
121
122 oss << std::setw(3)
123 << std::setfill('0')
124 << graus
125 << std::fixed
126 << std::setprecision(4)
127 << std::setw(7)
128 << std::setfill('0')
129 << min;
130 }
131
132 ddmm = oss.str();
133 }

363 GPSSim(

375 ){
376 throw std::runtime_error("Falha ao criar pseudo-terminal");
377 }
378
379 // Configura o terminal filho para simular o módulo real (9600 8N1)
380 termios config_com{}; // Cria a estrutura vazia
381 ::tcgetattr(fd_filho, &config_com); // Lê as configurações atuais e armazena na struct
382 ::cfsetispeed(&config_com, B9600); // Definimos velocidade de entrada e de saída
383 ::cfsetospeed(&config_com, B9600); // Essa constante está presente dentro do termios.h
384 // Diversas operações bits a bits
385 config_com.c_cflag = (config_com.c_cflag & ~CSIZE) | CS8;
386 config_com.c_cflag |= (CLOCAL | CREAD);
387 config_com.c_cflag &= ~(PARENB | CSTOPB);
388 config_com.c_iflag = IGNPAR;
389 config_com.c_oflag = 0;
390 config_com.c_lflag = 0;
391 tcsetattr(fd_filho, TCSANOW, &config_com); // Aplicamos as configurações
392
393 // Fecha o filho - será aberto pelo usuário no caminho correto
394 // Mantemos o Pai aberto para procedimentos posteriores
395 ::close(fd_filho);
396 }
