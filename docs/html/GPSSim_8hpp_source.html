115 static void
116 converter_graus_para_nmea(
117 double graus_decimais,
118 bool is_lat,

121 ){
122
123 hemisf = (is_lat) ? (
124 ( graus_decimais >= 0 ) ? 'N' : 'S'
125 ) :
126 (
127 ( graus_decimais >= 0 ) ? 'E' : 'W'
128 );
129
130 double valor_abs = std::fabs(graus_decimais);
131 int graus = static_cast<int>(std::floor(valor_abs));
132 double min = (valor_abs - graus) * 60.0;
133
134 // Não utilizamos apenas a função de formatar_inteiro, pois
135 // utilizaremos o oss em seguida.
136 std::ostringstream oss;
137 if(
138 is_lat
139 ){
140
141 oss << std::setw(2)
142 << std::setfill('0')
143 << graus
144 << std::fixed
145 << std::setprecision(4)
146 << std::setw(7)
147 << std::setfill('0')
148 << min;
149 }
150 else{
151
152 oss << std::setw(3)
153 << std::setfill('0')
154 << graus
155 << std::fixed
156 << std::setprecision(4)
157 << std::setw(7)
158 << std::setfill('0')
159 << min;
160 }
161
162 ddmm = oss.str();

170 static std::string
171 finalizar_corpo_de_frase(
172 const std::string& corpo_frase
173 ){
174
175 uint8_t paridade = 0;
176 for(
177 char caract : corpo_frase
178 ){
179
180 paridade ^= (uint8_t)caract;
181 }
182
183 std::ostringstream oss;
184 oss << '$'
185 << corpo_frase
186 << '*'
187 << std::uppercase
188 << std::hex
189 << std::setw(2)
190 << std::setfill('0')
191 << (int)paridade
192 << "\r\n";
193
194 return oss.str();

249 << lon_nmea << "," << hemisferio_lon << ","
250 << std::fixed << std::setprecision(2) << velocidade_nos << ",0.00,"
251 << formatar_inteiro(tempo_utc.tm_mday, 2)
252 << formatar_inteiro(tempo_utc.tm_mon + 1, 2)
253 << formatar_inteiro((tempo_utc.tm_year + 1900) % 100, 2)
254 << ",,,A";

256 return finalizar_corpo_de_frase(oss.str());

309 void
310 _update_position(){
311 if (!_mov_circ){ return; }
312
313 using namespace std::chrono;
314 double tempo_decorrido = duration<double>(steady_clock::now() - _start_time).count();
315 double angulo = (2.0 * M_PI) * std::fmod(tempo_decorrido / _periodo_circ, 1.0);
316
317 // Aproximação: 1° de latitude = 111320 metros
318 // Ajuste da longitude pelo cosseno da latitude
319 double delta_lat = (_raio * std::sin(angulo)) / 111320.0;
320 double delta_lon = (_raio * std::cos(angulo)) / (111320.0 * std::cos(_lat * M_PI / 180.0));
321
322 _lat_sim = _lat + delta_lat;
323 _lon_sim = _lon + delta_lon;
324 }

330 void
331 _loop(){
332 // Inicializa posição simulada
333 _lat_sim = _lat;
334 _lon_sim = _lon;
335
336 while (_is_exec){
337
338 // Gera frases NMEA
339 auto rmc = _gerar_corpo_de_frase_rmc(_lat_sim, _lon_sim, _velocidade_nos);
340 auto gga = _gerar_corpo_de_frase_gga(_lat_sim, _lon_sim, _alt, 10, 0.8);
341
342 const std::string saida = rmc + gga;
343 // std::cout << "\033[7mGPS6MV2 Simulado Emitindo:\033[0m \n" << saida << std::endl;
344 (void)!::write(_fd_pai, saida.data(), saida.size());
345
346 // Aguarda o próximo ciclo
347 std::this_thread::sleep_for(_periodo_atualizacao);
348
349 _update_position();
350 }

371 explicit
372 GPSSim(
373 double latitude_inicial_graus,
374 double longitude_inicial_graus,
375 double altitude_metros = 10.0,

383 std::chrono::milliseconds(1000)),

385 {
386
387 // Cria o par de pseudo-terminais
388 if(

390 ){
391 throw std::runtime_error("Falha ao criar pseudo-terminal");
392 }
393
394 // Configura o terminal filho para simular o módulo real (9600 8N1)
395 termios config_com{}; // Cria a estrutura vazia
396 ::tcgetattr(_fd_filho, &config_com); // Lê as configurações atuais e armazena na struct
397 ::cfsetispeed(&config_com, B115200); // Definimos velocidade de entrada e de saída
398 ::cfsetospeed(&config_com, B115200); // Essa constante está presente dentro do termios.h
399 // Diversas operações bits a bits
400 config_com.c_cflag = (config_com.c_cflag & ~CSIZE) | CS8;
401 config_com.c_cflag |= (CLOCAL | CREAD);
402 config_com.c_cflag &= ~(PARENB | CSTOPB);
403 config_com.c_iflag = IGNPAR;
404 config_com.c_oflag = 0;
405 config_com.c_lflag = 0;
406 tcsetattr(_fd_filho, TCSANOW, &config_com); // Aplicamos as configurações
407
408 // Fecha o filho - será aberto pelo usuário no caminho correto
409 // Mantemos o Pai aberto para procedimentos posteriores
410 ::close(_fd_filho);
411 }
